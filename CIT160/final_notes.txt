function golfScore(hand) {
var score = 0;
for (var i = 0; i < hand.length; i++) {
if (hand[i] === "King" || hand[i] === "Queen" || hand[i] === "Jack") {
score += 10;
} else {
score += hand[i];
}
}
return score;
}
